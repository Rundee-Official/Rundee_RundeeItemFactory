19 PlayerProfile LoadProfileFromPath(const std::string& filePath)
20 {
21 PlayerProfile profile;
22
23 if (!std::filesystem::exists(filePath))
24 {
25 return profile;
26 }
27
28 std::ifstream file(filePath);
29 if (!file.is_open())
30 {
31 return profile;
32 }
33
34 try
35 {
36 json j;
37 file >> j;
38
39 profile.id = j.value("id", "");
40 profile.displayName = j.value("displayName", "");
41 profile.description = j.value("description", "");
42 profile.version = j.value("version", 1);
43 profile.isDefault = j.value("isDefault", false);
44
45 // Load playerSettings
46 if (j.contains("playerSettings") && j["playerSettings"].is_object())
47 {
48 const auto& ps = j["playerSettings"];
49 profile.playerSettings.maxHunger = ps.value("maxHunger", 100);
50 profile.playerSettings.maxThirst = ps.value("maxThirst", 100);
51 profile.playerSettings.maxHealth = ps.value("maxHealth", 100);
52 profile.playerSettings.maxStamina = ps.value("maxStamina", 100);
53 profile.playerSettings.maxWeight = ps.value("maxWeight", 50000);
54 profile.playerSettings.maxEnergy = ps.value("maxEnergy", 100);
55 }
56
57 // Load statSections
58 if (j.contains("statSections") && j["statSections"].is_array())
59 {
60 for (const auto& sectionJson : j["statSections"])
61 {
62 PlayerStatSection section;
63 section.name = sectionJson.value("name", "");
64 section.displayName = sectionJson.value("displayName", "");
65 section.description = sectionJson.value("description", "");
66 section.displayOrder = sectionJson.value("displayOrder", 0);
67
68 if (sectionJson.contains("fields") && sectionJson["fields"].is_array())
69 {
70 for (const auto& fieldJson : sectionJson["fields"])
71 {
72 PlayerStatField field;
73 field.name = fieldJson.value("name", "");
74 field.displayName = fieldJson.value("displayName", "");
75 field.description = fieldJson.value("description", "");
76 field.value = fieldJson.value("value", 0);
77 field.displayOrder = fieldJson.value("displayOrder", 0);
78
79 section.fields.push_back(field);
80 }
81 }
82
83 profile.statSections.push_back(section);
84 }
85 }
86 }
87 catch (const std::exception& e)

89 std::cerr << "[PlayerProfileManager] Failed to parse profile: " << e.what() << "\n";
90 }

95 PlayerProfile LoadProfile(const std::string& profileId, const std::string& profilesDir)
96 {
97 if (profileId.empty() || profilesDir.empty())
98 {
99 return PlayerProfile();

101
102 std::filesystem::path profilePath(profilesDir);
103 profilePath /= (profileId + ".json");

108 PlayerProfile GetDefaultProfile(const std::string& profilesDir)
109 {
110 if (!std::filesystem::exists(profilesDir) || !std::filesystem::is_directory(profilesDir))
111 {
112 return PlayerProfile();
113 }
114
115 // Look for profile with isDefault = true
116 for (const auto& entry : std::filesystem::directory_iterator(profilesDir))
117 {
118 if (entry.is_regular_file() && entry.path().extension() == ".json")
119 {
120 PlayerProfile profile = LoadProfileFromPath(entry.path().string());
121 if (!profile.id.empty() && profile.isDefault)
122 {
123 return profile;
124 }
125 }
126 }
127
128 // If no default found, return first profile
129 for (const auto& entry : std::filesystem::directory_iterator(profilesDir))
130 {
131 if (entry.is_regular_file() && entry.path().extension() == ".json")
132 {
133 PlayerProfile profile = LoadProfileFromPath(entry.path().string());
134 if (!profile.id.empty())
135 {
136 return profile;
137 }
138 }
139 }
140
141 return PlayerProfile();
142 }
