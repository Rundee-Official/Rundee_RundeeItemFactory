54/**
55 * @brief Template implementation for AppendIdsToRegistry
56 *
57 * Loads existing IDs from registry, adds new IDs from items,
58 * saves back to registry, and logs the event.
59 *
60 * @tparam T Item type with an 'id' member (e.g., ItemData, nlohmann::json)
61 * @param typeName Item type name (e.g., "Food", "Weapon")
62 * @param items Vector of items to extract IDs from
63 */
64template <typename T>
65void ItemGeneratorRegistry::AppendIdsToRegistry(const std::string& typeName, const std::vector<T>& items)
66{
67 if (items.empty())
68 return;
69
70 std::set<std::string> ids = LoadRegistryIds(typeName);
71 size_t before = ids.size();
72 for (const auto& item : items)
73 {
74 ids.insert(item.id);
75 }
76 size_t after = ids.size();
77 size_t added = (after >= before) ? (after - before) : 0;
78 if (SaveRegistryIds(typeName, ids))
79 {
80 LogRegistryEvent(typeName, before, added, after);
81 }
82}
