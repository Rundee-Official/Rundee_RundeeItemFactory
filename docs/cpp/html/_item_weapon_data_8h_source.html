18/**
19 * @struct WeaponAttachmentSlot
20 * @brief Defines an attachment slot on a weapon
21 *
22 * Represents a single slot where a weapon component can be attached.
23 * Supports multiple slots of the same type (e.g., multiple rail slots).
24 */
25struct WeaponAttachmentSlot
26{
27 /**
28 * @brief Slot type identifier
29 *
30 * Examples: "Muzzle", "Grip", "Sight", "Scope", "Stock", "Barrel",
31 * "Handguard", "Rail", etc. Must match componentType of compatible components.
32 */
33 std::string slotType;
34
35 /**
36 * @brief Slot index for multiple slots of same type
37 *
38 * Used when a weapon has multiple slots of the same type (e.g., multiple rail slots).
39 * Index 0 is the first slot, 1 is the second, etc.
40 */
41 int slotIndex = 0;
42
43 /**
44 * @brief Whether this slot must be filled
45 *
46 * If true, the weapon cannot function without a component in this slot.
47 */
48 bool isRequired = false;
49};
