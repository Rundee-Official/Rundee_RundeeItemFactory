19 void Validate(ItemAmmoData& item)
20 {
21 std::string originalId = item.id;
22
23 // Normalize prefix: strip any leading ammo_ (any case, repeated), then add canonical "Ammo_"
24 if (!item.id.empty())
25 {
26 auto stripPrefix = [](std::string& value, const std::string& prefixLower)
27 {
28 std::string lower = value;
29 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
30 while (lower.rfind(prefixLower, 0) == 0)
31 {
32 value = value.substr(prefixLower.size());
33 lower = value;
34 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
35 }
36 };
37
38 stripPrefix(item.id, "ammo_");
39 item.id = "Ammo_" + item.id;
40 }
41
42 // Debug logging disabled for release
43
44 // Default category
45 if (item.category.empty())
46 item.category = "Ammo";
47
48 // Clamp stat values
49 item.damageBonus = JsonUtils::ClampInt(item.damageBonus, -50, 50);
50 item.penetration = JsonUtils::ClampInt(item.penetration, 0, 100);
51 item.accuracyBonus = JsonUtils::ClampInt(item.accuracyBonus, -50, 50);
52 item.recoilModifier = JsonUtils::ClampInt(item.recoilModifier, -50, 50);
53 item.value = JsonUtils::ClampInt(item.value, 0, 100);
54 item.maxStack = JsonUtils::ClampInt(item.maxStack, 1, 999);
55
56 // Special property logic
57 if (item.armorPiercing)
58 {
59 // Armor piercing ammo should have high penetration
60 if (item.penetration < 50)
61 item.penetration = 50;
62 }
63
64 if (item.hollowPoint)
65 {
66 // Hollow point has higher damage but lower penetration
67 if (item.damageBonus < 5)
68 item.damageBonus = 5;
69 if (item.penetration > 30)
70 item.penetration = 30;
71 }
72
73 // Ensure description is not empty
74 if (item.description.empty())
75 {
76 item.description = "A " + item.displayName + " ammunition.";
77 std::cout << "[AmmoItemValidator] Warning: Item " << item.id << " has empty description, using default.\n";
78 }
79 }
80}
