15std::string DynamicPromptBuilder::BuildPromptFromProfile(
16 const ItemProfile& profile,
17 const PlayerProfile& playerProfile,
18 const FoodGenerateParams& params,
19 const std::set<std::string>& existingIds,
20 const std::string& modelName,
21 const std::string& generationTimestamp,
22 int existingCount)
23{
24 std::ostringstream prompt;
25
26 // Try to load template first (if templates are used in the future)
27 std::string templateName = profile.itemTypeName;
28 std::transform(templateName.begin(), templateName.end(), templateName.begin(), ::tolower);
29 templateName += "_template";
30
31 std::string worldContext = profile.customContext.empty() ? "" : profile.customContext;
32 std::string templateContent = PromptTemplateLoader::LoadTemplate(
33 templateName,
34 worldContext,
35 params.maxHunger,
36 params.maxThirst,
37 params.count,
38 existingIds,
39 profile.displayName, // Used as presetName replacement (display name of profile)
40 profile.itemTypeName,
41 modelName,
42 generationTimestamp,
43 existingCount);
44
45 // If template exists, use it as base
46 if (!templateContent.empty())
47 {
48 prompt << templateContent;
49 }
50 else
51 {
52 // Build prompt from scratch
53 prompt << "You are a game item generator. Generate " << params.count
54 << " unique " << profile.itemTypeName << " items as a JSON array.\n\n";
55
56 // Use Item Profile's customContext as world context (Preset system removed)
57 if (!profile.customContext.empty())
58 {
59 prompt << "World Context:\n" << profile.customContext << "\n\n";
60 }
61 }
62
63 // Add profile information section
64 prompt << "\n=== ITEM PROFILE SPECIFICATION ===\n\n";
65 prompt << "Profile: " << profile.displayName << " (" << profile.id << ")\n";
66 if (!profile.description.empty())
67 {
68 prompt << "Description: " << profile.description << "\n";
69 }
70 prompt << "Item Type: " << profile.itemTypeName << "\n";
71
72 // Note: customContext is already included in "World Context" section above, so we don't duplicate it here
73
74 // Add metadata if provided
75 if (!profile.metadata.empty())
76 {
77 prompt << "Metadata: ";
78 bool first = true;
79 for (const auto& [key, value] : profile.metadata)
80 {
81 if (!first) prompt << ", ";
82 prompt << key << ": " << value.dump();
83 first = false;
84 }
85 prompt << "\n";
86 }
87 prompt << "\n";
88
89 // Add player profile information
90 prompt << "\n=== PLAYER PROFILE SPECIFICATION ===\n\n";
91 prompt << "Player Profile: " << playerProfile.displayName << " (" << playerProfile.id << ")\n";
92 if (!playerProfile.description.empty())
93 {
94 prompt << "Description: " << playerProfile.description << "\n";
95 }
96 prompt << "\n";
97
98 // Add player settings for balance context
99 prompt << "Player Stat Maximums (for balancing):\n";
100 prompt << " - Max Hunger: " << params.maxHunger << "\n";
101 prompt << " - Max Thirst: " << params.maxThirst << "\n";
102 prompt << " - Max Health: " << params.maxHealth << "\n";
103 prompt << " - Max Stamina: " << params.maxStamina << "\n";
104 prompt << " - Max Weight: " << params.maxWeight << " grams\n";
105 prompt << " - Max Energy: " << params.maxEnergy << "\n";
106 prompt << "\n";
107
108 // Add player stat sections (custom stat fields)
109 if (!playerProfile.statSections.empty())
110 {
111 prompt << "Player Stat Sections (Additional Context):\n";
112
113 // Sort sections by displayOrder
114 std::vector<PlayerStatSection> sortedSections = playerProfile.statSections;
115 std::sort(sortedSections.begin(), sortedSections.end(),
116 [](const PlayerStatSection& a, const PlayerStatSection& b) {
117 return a.displayOrder < b.displayOrder;
118 });
119
120 for (const auto& section : sortedSections)
121 {
122 prompt << "\n Section: " << section.displayName;
123 if (!section.name.empty() && section.name != section.displayName)
124 {
125 prompt << " (" << section.name << ")";
126 }
127 prompt << "\n";
128
129 if (!section.description.empty())
130 {
131 prompt << " Description: " << section.description << "\n";
132 }
133
134 if (!section.fields.empty())
135 {
136 // Sort fields by displayOrder
137 std::vector<PlayerStatField> sortedFields = section.fields;
138 std::sort(sortedFields.begin(), sortedFields.end(),
139 [](const PlayerStatField& a, const PlayerStatField& b) {
140 return a.displayOrder < b.displayOrder;
141 });
142
143 prompt << " Fields:\n";
144 for (const auto& field : sortedFields)
145 {
146 prompt << " - " << field.displayName;
147 if (!field.name.empty() && field.name != field.displayName)
148 {
149 prompt << " (" << field.name << ")";
150 }
151 prompt << ": " << field.value;
152 if (!field.description.empty())
153 {
154 prompt << " - " << field.description;
155 }
156 prompt << "\n";
157 }
158 }
159 }
160 prompt << "\n";
161 }
162
163 // Add all field definitions with validation rules
164 prompt << "=== REQUIRED FIELDS AND VALIDATION RULES ===\n\n";
165 prompt << "CRITICAL: Every item MUST have 'id' and 'displayName' fields. These are ALWAYS required and must be generated FIRST.\n";
166 prompt << "- 'displayName': Human-readable name that clearly identifies the item (e.g., 'AK-47 Assault Rifle', 'FN SCAR-17S Enhanced', 'Healing Potion').\n";
167 prompt << "- 'id': Unique identifier based on displayName in format {itemType}_{cleanedDisplayName} (e.g., weapon_ak47assaultrifle, weapon_fnscar17senhanced, food_healingpotion).\n";
168 prompt << " The ID should be derived from the displayName by:\n";
169 prompt << " 1. Converting to lowercase\n";
170 prompt << " 2. Removing spaces, hyphens, and special characters\n";
171 prompt << " 3. Keeping only alphanumeric characters\n";
172 prompt << " 4. Prefixing with item type (e.g., 'weapon_', 'food_')\n";
173 prompt << " Example: 'FN SCAR-17S Enhanced' -> 'weapon_fnscar17senhanced'\n\n";
174 prompt << "Each item MUST include the following fields with these specifications:\n\n";
175
176 // Sort fields by display order
177 std::vector<ProfileField> sortedFields = profile.fields;
178 std::sort(sortedFields.begin(), sortedFields.end(),
179 [](const ProfileField& a, const ProfileField& b) {
180 return a.displayOrder < b.displayOrder;
181 });
182
183 // Emphasize id and displayName fields
184 for (const auto& field : sortedFields)
185 {
186 // Special emphasis for id and displayName
187 if (field.name == "id" || field.name == "displayName")
188 {
189 prompt << "*** CRITICAL FIELD ***\n";
190 }
191 prompt << "Field: " << field.name << "\n";
192 prompt << " Display Name: " << field.displayName << "\n";
193 prompt << " Description: " << field.description << "\n";
194 prompt << " Category: " << field.category << "\n";
195 prompt << " Type: ";
196
197 switch (field.type)
198 {
199 case ProfileFieldType::String:
200 prompt << "string";
201 break;
202 case ProfileFieldType::Integer:
203 prompt << "integer";
204 break;
205 case ProfileFieldType::Float:
206 prompt << "float";
207 break;
208 case ProfileFieldType::Boolean:
209 prompt << "boolean";
210 break;
211 case ProfileFieldType::Array:
212 prompt << "array";
213 break;
214 case ProfileFieldType::Object:
215 prompt << "object";
216 break;
217 }
218 prompt << "\n";
219
220 // Validation rules
221 if (field.validation.isRequired)
222 {
223 prompt << " REQUIRED: Yes\n";
224 }
225 else
226 {
227 prompt << " REQUIRED: No";
228 if (!field.defaultValue.is_null())
229 {
230 prompt << " (default: " << field.defaultValue.dump() << ")";
231 }
232 prompt << "\n";
233 }
234
235 // Type-specific validation
236 if (field.type == ProfileFieldType::String)
237 {
238 if (field.validation.minLength > 0)
239 prompt << " Min Length: " << field.validation.minLength << "\n";
240 if (field.validation.maxLength > 0)
241 prompt << " Max Length: " << field.validation.maxLength << "\n";
242 if (!field.validation.allowedValues.empty())
243 {
244 prompt << " Allowed Values: ";
245 for (size_t i = 0; i < field.validation.allowedValues.size(); ++i)
246 {
247 if (i > 0) prompt << ", ";
248 prompt << field.validation.allowedValues[i];
249 }
250 prompt << "\n";
251 }
252 }
253 else if (field.type == ProfileFieldType::Integer || field.type == ProfileFieldType::Float)
254 {
255 if (field.validation.minValue != 0.0)
256 prompt << " Min Value: " << field.validation.minValue << "\n";
257 if (field.validation.maxValue != 0.0)
258 prompt << " Max Value: " << field.validation.maxValue << "\n";
259 }
260 else if (field.type == ProfileFieldType::Array)
261 {
262 if (field.validation.minLength > 0)
263 prompt << " Min Elements: " << field.validation.minLength << "\n";
264 if (field.validation.maxLength > 0)
265 prompt << " Max Elements: " << field.validation.maxLength << "\n";
266 }
267
268 // Relationship constraints
269 if (!field.validation.relationshipConstraints.empty())
270 {
271 prompt << " Relationship Constraints:\n";
272 for (const auto& constraint : field.validation.relationshipConstraints)
273 {
274 prompt << " - " << constraint.description << "\n";
275 prompt << " (" << field.name << " " << constraint.operator_
276 << " " << constraint.targetField << ")\n";
277 }
278 }
279
280 // Custom constraint
281 if (!field.validation.customConstraint.empty())
282 {
283 prompt << " Custom Constraint: " << field.validation.customConstraint << "\n";
284 }
285
286 prompt << "\n";
287 }
288
289 // Add existing IDs to avoid
290 if (!existingIds.empty())
291 {
292 prompt << "\n=== EXISTING ITEM IDs TO AVOID ===\n";
293 prompt << "IMPORTANT: Do NOT use these existing item IDs. Generate NEW unique IDs.\n";
294 prompt << "Avoid reusing stems; use fresh, novel names, not simple number suffixes.\n\n";
295
296 int idCount = 0;
297 for (const auto& id : existingIds)
298 {
299 if (idCount > 0) prompt << ", ";
300 prompt << id;
301 idCount++;
302 if (idCount >= 20) // Limit to first 20
303 {
304 prompt << " ... (and " << (existingIds.size() - 20) << " more)";
305 break;
306 }
307 }
308 prompt << "\n\n";
309 }
310
311 // Output format instructions
312 prompt << "\n=== OUTPUT FORMAT ===\n";
313 prompt << "Return a JSON array of " << params.count << " items.\n";
314 prompt << "Each item must be a JSON object with all required fields.\n";
315 prompt << "Example structure:\n";
316 prompt << "[\n";
317 prompt << " {\n";
318
319 // Show example with first few fields
320 int exampleCount = 0;
321 for (const auto& field : sortedFields)
322 {
323 if (exampleCount >= 5) break;
324 prompt << " \"" << field.name << "\": ";
325 switch (field.type)
326 {
327 case ProfileFieldType::String:
328 prompt << "\"example_value\"";
329 break;
330 case ProfileFieldType::Integer:
331 prompt << "0";
332 break;
333 case ProfileFieldType::Float:
334 prompt << "0.0";
335 break;
336 case ProfileFieldType::Boolean:
337 prompt << "false";
338 break;
339 case ProfileFieldType::Array:
340 prompt << "[]";
341 break;
342 case ProfileFieldType::Object:
343 prompt << "{}";
344 break;
345 }
346 prompt << ",\n";
347 exampleCount++;
348 }
349 prompt << " ...\n";
350 prompt << " }\n";
351 prompt << "]\n\n";
352
353 prompt << "Generate " << params.count << " unique, creative " << profile.itemTypeName
354 << " items that fit the world context and follow all validation rules.\n";
355
356 return prompt.str();
