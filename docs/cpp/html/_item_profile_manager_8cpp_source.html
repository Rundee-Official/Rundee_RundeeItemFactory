166bool ItemProfileManager::ValidateProfile(const ItemProfile& profile, std::vector<std::string>& errors)
167{
168 errors.clear();
169
170 if (profile.id.empty())
171 {
172 errors.push_back("Profile ID is required");
173 }
174
175 if (profile.itemTypeName.empty())
176 {
177 errors.push_back("Item type name is required");
178 }
179
180 if (profile.fields.empty())
181 {
182 errors.push_back("Profile must have at least one field");
183 }
184
185 // Check for duplicate field names
186 std::set<std::string> fieldNames;
187 for (const auto& field : profile.fields)
188 {
189 if (fieldNames.count(field.name))
190 {
191 errors.push_back("Duplicate field name: " + field.name);
192 }
193 fieldNames.insert(field.name);
194 }
195
196 return errors.empty();
197}

357void ItemProfileManager::EnsureRequiredFields(ItemProfile& profile)
358{
359 // Check if id field exists
360 bool hasId = false;
361 bool hasDisplayName = false;
362 int minDisplayOrder = INT_MAX;
363
364 for (const auto& field : profile.fields)
365 {
366 if (field.name == "id")
367 hasId = true;
368 if (field.name == "displayName")
369 hasDisplayName = true;
370 if (field.displayOrder < minDisplayOrder)
371 minDisplayOrder = field.displayOrder;
372 }
373
374 // If no fields exist, start from 0
375 if (minDisplayOrder == INT_MAX)
376 minDisplayOrder = 0;
377
378 // Add id field if missing (at position 0)
379 if (!hasId)
380 {
381 ProfileField idField;
382 idField.name = "id";
383 idField.type = ProfileFieldType::String;
384 idField.displayName = "ID";
385 idField.description = "Unique identifier for this item. Must be unique across all items. Format: {itemType}_{number} (e.g., weapon_001, food_042)";
386 idField.category = "Basic";
387 idField.displayOrder = minDisplayOrder - 2; // Ensure it's first
388 idField.defaultValue = nlohmann::json();
389 idField.validation.isRequired = true;
390 idField.validation.minLength = 1;
391 idField.validation.maxLength = 100;
392 profile.fields.insert(profile.fields.begin(), idField);
393 }
394 else
395 {
396 // Ensure id field is required and has proper description
397 for (auto& field : profile.fields)
398 {
399 if (field.name == "id")
400 {
401 field.validation.isRequired = true;
402 if (field.description.empty())
403 {
404 field.description = "Unique identifier for this item. Must be unique across all items. Format: {itemType}_{number} (e.g., weapon_001, food_042)";
405 }
406 // Move to first position
407 if (field.displayOrder != minDisplayOrder - 2)
408 {
409 field.displayOrder = minDisplayOrder - 2;
410 }
411 break;
412 }
413 }
414 }
415
416 // Add displayName field if missing (at position 1)
417 if (!hasDisplayName)
418 {
419 ProfileField displayNameField;
420 displayNameField.name = "displayName";
421 displayNameField.type = ProfileFieldType::String;
422 displayNameField.displayName = "Display Name";
423 displayNameField.description = "Human-readable name for this item. Should clearly identify what the item is (e.g., 'AK-47 Assault Rifle', 'Healing Potion')";
424 displayNameField.category = "Basic";
425 displayNameField.displayOrder = minDisplayOrder - 1; // Second after id
426 displayNameField.defaultValue = nlohmann::json();
427 displayNameField.validation.isRequired = true;
428 displayNameField.validation.minLength = 1;
429 displayNameField.validation.maxLength = 200;
430 // Insert after id field
431 if (profile.fields.size() > 0 && profile.fields[0].name == "id")
432 {
433 profile.fields.insert(profile.fields.begin() + 1, displayNameField);
434 }
435 else
436 {
437 profile.fields.insert(profile.fields.begin(), displayNameField);
438 }
439 }
440 else
441 {
442 // Ensure displayName field is required and has proper description
443 for (auto& field : profile.fields)
444 {
445 if (field.name == "displayName")
446 {
447 field.validation.isRequired = true;
448 if (field.description.empty())
449 {
450 field.description = "Human-readable name for this item. Should clearly identify what the item is (e.g., 'AK-47 Assault Rifle', 'Healing Potion')";
451 }
452 // Move to second position
453 if (field.displayOrder != minDisplayOrder - 1)
454 {
455 field.displayOrder = minDisplayOrder - 1;
456 }
457 break;
458 }
459 }
460 }
461
462 // Sort fields by displayOrder to ensure id and displayName are first
463 std::sort(profile.fields.begin(), profile.fields.end(),
464 [](const ProfileField& a, const ProfileField& b) {
465 return a.displayOrder < b.displayOrder;
466 });
467}

469ItemProfile ItemProfileManager::ParseProfileFromJson(const nlohmann::json& json)
470{
471 ItemProfile profile;
472
473 if (json.contains("id") && json["id"].is_string())
474 profile.id = json["id"];
475 if (json.contains("displayName") && json["displayName"].is_string())
476 profile.displayName = json["displayName"];
477 if (json.contains("description") && json["description"].is_string())
478 profile.description = json["description"];
479 if (json.contains("itemTypeName") && json["itemTypeName"].is_string())
480 profile.itemTypeName = json["itemTypeName"];
481 if (json.contains("version") && json["version"].is_number())
482 profile.version = json["version"];
483 if (json.contains("isDefault") && json["isDefault"].is_boolean())
484 profile.isDefault = json["isDefault"];
485 if (json.contains("customContext") && json["customContext"].is_string())
486 profile.customContext = json["customContext"];
487
488 if (json.contains("fields") && json["fields"].is_array())
489 {
490 for (const auto& fieldJson : json["fields"])
491 {
492 profile.fields.push_back(ParseFieldFromJson(fieldJson));
493 }
494 }
495
496 // Ensure id and displayName fields are always present at the top
497 EnsureRequiredFields(profile);
498
499 // Player settings removed from ItemProfile - now using PlayerProfile instead
500 // Legacy support: ignore playerSettings if present in JSON (for backward compatibility)
501
502 return profile;
503}

617
618 if (json.contains("allowedValues") && json["allowedValues"].is_array())
619 {
620 for (const auto& val : json["allowedValues"])
621 {
622 if (val.is_string())
623 validation.allowedValues.push_back(val.get<std::string>());
624 }
