171 ItemType ParseItemType(const std::string& typeStr)
172 {
173 std::string lower = typeStr;
174 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
175
176 if (lower == "food")
177 return ItemType::Food;
178 else if (lower == "drink")
179 return ItemType::Drink;
180 else if (lower == "material")
181 return ItemType::Material;
182 else if (lower == "weapon")
183 return ItemType::Weapon;
184 else if (lower == "weaponcomponent" || lower == "weapon_component")
185 return ItemType::WeaponComponent;
186 else if (lower == "ammo")
187 return ItemType::Ammo;
188 else
189 return ItemType::Food; // Default
190 }

192 std::vector<BatchItem> ParseBatchString(const std::string& batchStr)
193 {
194 std::vector<BatchItem> result;
195
196 if (batchStr.empty())
197 {
198 std::cerr << "[CommandLineParser] Error: Batch string is empty.\n";
199 return result;
200 }
201
202 std::istringstream iss(batchStr);
203 std::string token;
204
205 while (std::getline(iss, token, ','))
206 {
207 // Trim whitespace
208 token.erase(0, token.find_first_not_of(" \t"));
209 token.erase(token.find_last_not_of(" \t") + 1);
210
211 if (token.empty())
212 continue;
213
214 // Parse format: "itemtype:count" or "itemtype:count:outputpath"
215 size_t colon1 = token.find(':');
216 if (colon1 == std::string::npos)
217 {
218 std::cerr << "[CommandLineParser] Warning: Invalid batch item format: " << token
219 << " (expected 'itemtype:count' or 'itemtype:count:outputpath')\n";
220 continue;
221 }
222
223 std::string typeStr = token.substr(0, colon1);
224 std::string rest = token.substr(colon1 + 1);
225
226 size_t colon2 = rest.find(':');
227 std::string countStr = (colon2 == std::string::npos) ? rest : rest.substr(0, colon2);
228 std::string outputPath = (colon2 == std::string::npos) ? "" : rest.substr(colon2 + 1);
229
230 ItemType itemType = ParseItemType(typeStr);
231 int count = std::atoi(countStr.c_str());
232
233 if (count <= 0)
234 {
235 std::cerr << "[CommandLineParser] Warning: Invalid count for " << typeStr
236 << ": " << countStr << "\n";
237 continue;
238 }
239
240 BatchItem batchItem;
241 batchItem.itemType = itemType;
242 batchItem.count = count;
243 batchItem.outputPath = outputPath;
244
245 result.push_back(batchItem);
246 }
247
248 if (result.empty())
249 {
250 std::cerr << "[CommandLineParser] Error: No valid batch items parsed from: " << batchStr << "\n";
251 }
252 else
253 {
254 std::cout << "[CommandLineParser] Parsed " << result.size() << " batch items.\n";
255 }
256
257 return result;
258 }
