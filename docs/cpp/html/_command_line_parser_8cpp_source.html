24 * @brief Get the directory where the executable is located
25 * @return Executable directory path (with trailing slash/backslash)
26 */
27 static std::string GetExecutableDirectory()
28 {
29#ifdef _WIN32
30 char exePath[MAX_PATH];
31 DWORD pathLen = GetModuleFileNameA(NULL, exePath, MAX_PATH);
32 if (pathLen == 0 || pathLen >= MAX_PATH)
33 {
34 return ""; // Fallback to relative path
35 }
36
37 // Extract directory
38 std::string exeDir = exePath;
39 size_t lastSlash = exeDir.find_last_of("\\/");
40 if (lastSlash != std::string::npos)
41 {
42 exeDir = exeDir.substr(0, lastSlash + 1);
43 }
44 return exeDir;
45#else
46 // For non-Windows, use current directory as fallback
47 return "";
48#endif
49 }

51 CommandLineArgs ParseArguments(int argc, char** argv)
52 {
53 CommandLineArgs args;
54 args.params.count = 5;
55 args.params.maxHunger = 100;
56 args.params.maxThirst = 100;
57
58 // Get executable directory and set default output path relative to it
59 std::string exeDir = GetExecutableDirectory();
60 args.params.outputPath = exeDir + "ItemJson/items_food.json";
61
62 for (int i = 1; i < argc; ++i)
63 {
64 std::string arg = argv[i];
65
66 if (arg == "--model" && i + 1 < argc)
67 {
68 args.modelName = argv[++i];
69 }
70 else if (arg == "--mode" && i + 1 < argc)
71 {
72 std::string m = argv[++i];
73 if (m == "llm")
74 {
75 args.mode = RunMode::LLM;
76 }
77 else
78 {
79 std::cout << "[Warning] Unknown mode: " << m
80 << " (use 'llm')\n";
81 }
82 }
83 else if (arg == "--preset" && i + 1 < argc)
84 {
85 args.presetName = argv[++i];
86 }
87 else if (arg == "--itemType" && i + 1 < argc)
88 {
89 std::string t = argv[++i];
90 if (t == "food")
91 args.itemType = ItemType::Food;
92 else if (t == "drink")
93 args.itemType = ItemType::Drink;
94 else if (t == "medicine")
95 args.itemType = ItemType::Medicine;
96 else if (t == "material")
97 args.itemType = ItemType::Material;
98 else if (t == "weapon")
99 args.itemType = ItemType::Weapon;
100 else if (t == "weaponcomponent")
101 args.itemType = ItemType::WeaponComponent;
102 else if (t == "ammo")
103 args.itemType = ItemType::Ammo;
104 else if (t == "armor")
105 args.itemType = ItemType::Armor;
106 else if (t == "clothing")
107 args.itemType = ItemType::Clothing;
108 else
109 std::cout << "[Warning] Unknown itemType: " << t
110 << " (use 'food', 'drink', 'medicine', 'material', 'weapon', 'weaponcomponent', 'ammo', 'armor', or 'clothing')\n";
111 }
112 else if (arg == "--count" && i + 1 < argc)
113 {
114 args.params.count = std::atoi(argv[++i]);
115 }
116 else if (arg == "--maxHunger" && i + 1 < argc)
117 {
118 args.params.maxHunger = std::atoi(argv[++i]);
119 }
120 else if (arg == "--maxThirst" && i + 1 < argc)
121 {
122 args.params.maxThirst = std::atoi(argv[++i]);
123 }
124 else if (arg == "--maxHealth" && i + 1 < argc)
125 {
126 args.params.maxHealth = std::atoi(argv[++i]);
127 }
128 else if (arg == "--maxStamina" && i + 1 < argc)
129 {
130 args.params.maxStamina = std::atoi(argv[++i]);
131 }
132 else if (arg == "--maxWeight" && i + 1 < argc)
133 {
134 args.params.maxWeight = std::atoi(argv[++i]);
135 }
136 else if (arg == "--maxEnergy" && i + 1 < argc)
137 {
138 args.params.maxEnergy = std::atoi(argv[++i]);
139 }
140 else if (arg == "--out" && i + 1 < argc)
141 {
142 std::string outPath = argv[++i];
143 std::string exeDir = GetExecutableDirectory();
144
145 // Always save to .exe directory/ItemJson folder
146 // Extract filename from path (handles both absolute and relative)
147 std::filesystem::path pathObj(outPath);
148 std::string fileName = pathObj.filename().string();
149
150 // Ensure ItemJson folder path
151 std::filesystem::path itemJsonPath(exeDir);
152 itemJsonPath /= "ItemJson";
153 itemJsonPath /= fileName;
154
155 args.params.outputPath = itemJsonPath.string();
156 }
157 else if (arg == "--report" && i + 1 < argc)
158 {
159 args.reportPath = argv[++i];
160 // report mode is independent of LLM; mode value unused when reportPath is set
161 }
162 else if (arg == "--customPreset" && i + 1 < argc)
163 {
164 args.customPresetPath = argv[++i];
165 }
166 else if (arg == "--additionalPrompt" && i + 1 < argc)
167 {
168 args.additionalPrompt = argv[++i];
169 }
170 else if (arg == "--test" || arg == "--testMode")
171 {
172 args.useTestMode = true;
173 }
174 else if (arg == "--profile" && i + 1 < argc)
175 {
176 args.profileId = argv[++i];
177 }
178 else if (arg == "--playerProfile" && i + 1 < argc)
179 {
180 args.playerProfileId = argv[++i];
181 }
182 else
183 {
184 std::cout << "[Warning] Unknown or incomplete argument: " << arg << "\n";
185 }
186 }
187
188 return args;
189 }

217 ItemType ParseItemType(const std::string& typeStr)
218 {
219 std::string lower = typeStr;
220 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
221
222 if (lower == "food")
223 return ItemType::Food;
224 else if (lower == "drink")
225 return ItemType::Drink;
226 else if (lower == "medicine")
227 return ItemType::Medicine;
228 else if (lower == "material")
229 return ItemType::Material;
230 else if (lower == "weapon")
231 return ItemType::Weapon;
232 else if (lower == "weaponcomponent" || lower == "weapon_component")
233 return ItemType::WeaponComponent;
234 else if (lower == "ammo")
235 return ItemType::Ammo;
236 else if (lower == "armor")
237 return ItemType::Armor;
238 else if (lower == "clothing")
239 return ItemType::Clothing;
240 else
241 return ItemType::Food; // Default
242 }
