168 static void PrintFoodReport(const std::vector<ItemFoodData>& items)
169 {
170 if (items.empty())
171 {
172 std::cout << "[BalanceReporter] No food items to analyze.\n";
173 return;
174 }
175
176 FoodStats stats;
177 for (const auto& item : items)
178 {
179 stats.count++;
180 stats.totalHunger += item.hungerRestore;
181 stats.totalThirst += item.thirstRestore;
182 stats.totalHealth += item.healthRestore;
183
184 stats.minHunger = std::min(stats.minHunger, item.hungerRestore);
185 stats.maxHunger = std::max(stats.maxHunger, item.hungerRestore);
186 stats.minThirst = std::min(stats.minThirst, item.thirstRestore);
187 stats.maxThirst = std::max(stats.maxThirst, item.thirstRestore);
188 stats.minHealth = std::min(stats.minHealth, item.healthRestore);
189 stats.maxHealth = std::max(stats.maxHealth, item.healthRestore);
190
191 if (item.spoils)
192 stats.spoilingCount++;
193
194 stats.rarityCount[item.rarity]++;
195 }
196
197 std::cout << "\n";
198 std::cout << "========================================\n";
199 std::cout << " FOOD ITEMS BALANCE REPORT\n";
200 std::cout << "========================================\n";
201 std::cout << "Total Items: " << stats.count << "\n\n";
202
203 // Quick Summary
204 std::cout << "--- Quick Summary ---\n";
205 std::cout << " Average Hunger Restore: " << (stats.totalHunger / stats.count) << "\n";
206 std::cout << " Average Thirst Restore: " << (stats.totalThirst / stats.count) << "\n";
207 std::cout << " Average Health Restore: " << (stats.totalHealth / stats.count) << "\n";
208 float avgPower = (stats.totalHunger + stats.totalThirst + stats.totalHealth) / (float)stats.count;
209 std::cout << " Average Total Power: " << std::fixed << std::setprecision(1) << avgPower << "\n";
210 std::cout << " Spoiling Items: " << stats.spoilingCount << " ("
