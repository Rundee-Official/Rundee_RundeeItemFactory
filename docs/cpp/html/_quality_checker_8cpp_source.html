25 QualityResult CheckWeaponQuality(const ItemWeaponData& item)
26 {
27 QualityResult result;
28 result.isValid = true;
29 result.qualityScore = 100.0f;
30
31 // Check for unrealistic damage values
32 if (item.weaponCategory == "Ranged")
33 {
34 float avgDamage = (item.minDamage + item.maxDamage) / 2.0f;
35
36 // Ranged weapons: base damage should be reasonable (ammo adds bonus)
37 if (avgDamage > 80)
38 {
39 result.warnings.push_back("Very high base damage for ranged weapon (damage should come from ammo)");
40 result.qualityScore -= 15.0f;
41 }
42 else if (avgDamage < 10 && item.rarity != "Common")
43 {
44 result.warnings.push_back("Low base damage for non-common weapon");
45 result.qualityScore -= 10.0f;
46 }
47
48 // Check fire rate vs damage balance
49 if (item.fireRate > 800 && avgDamage > 50)
50 {
51 result.warnings.push_back("High fire rate combined with high damage may be overpowered");
52 result.qualityScore -= 10.0f;
53 }
54
55 // Check weight vs fire rate (heavy weapons usually have lower fire rate)
56 if (item.weight > 4000 && item.fireRate > 600)
57 {
58 result.warnings.push_back("Heavy weapon with high fire rate may be unrealistic");
59 result.qualityScore -= 5.0f;
60 }
61
62 // Check muzzle velocity vs caliber
63 if (!item.caliber.empty())
64 {
65 if (item.caliber.find("9mm") != std::string::npos && item.muzzleVelocity > 500)
66 {
67 result.warnings.push_back("9mm muzzle velocity seems high (typical: 300-400 m/s)");
68 result.qualityScore -= 5.0f;
69 }
70 else if (item.caliber.find("7.62") != std::string::npos && item.muzzleVelocity < 600)
71 {
72 result.warnings.push_back("7.62mm muzzle velocity seems low (typical: 700-900 m/s)");
73 result.qualityScore -= 5.0f;
74 }
75 }
76
77 // Check accuracy vs recoil balance
78 if (item.accuracy > 80 && item.recoil > 50)
79 {
80 result.warnings.push_back("High accuracy with high recoil is contradictory");
81 result.qualityScore -= 10.0f;
82 }
83 }
84 else if (item.weaponCategory == "Melee")
85 {
86 float avgDamage = (item.minDamage + item.maxDamage) / 2.0f;
87
88 // Melee weapons: damage should be inherent
89 if (avgDamage > 70)
90 {
91 result.warnings.push_back("Very high damage for melee weapon");
92 result.qualityScore -= 10.0f;
93 }
94
95 // Check attack speed vs damage balance
96 if (item.attackSpeed > 5 && avgDamage > 40)
97 {
98 result.warnings.push_back("High attack speed with high damage may be overpowered");
99 result.qualityScore -= 10.0f;
100 }
101
102 // Check reach vs weapon type
103 if (item.weaponType.find("Knife") != std::string::npos && item.reach > 100)
104 {
105 result.warnings.push_back("Knife with long reach is unrealistic");
106 result.qualityScore -= 10.0f;
107 }
108 else if (item.weaponType.find("Spear") != std::string::npos && item.reach < 200)
109 {
110 result.warnings.push_back("Spear with short reach is unrealistic");
111 result.qualityScore -= 5.0f;
112 }
113 }
114
115 // Check rarity vs stats correlation
116 if (item.rarity == "Common" && (item.minDamage + item.maxDamage) / 2.0f > 50)
117 {
118 result.warnings.push_back("Common weapon with high damage may be unbalanced");
119 result.qualityScore -= 5.0f;
120 }
121 else if (item.rarity == "Rare" && (item.minDamage + item.maxDamage) / 2.0f < 30)
122 {
123 result.warnings.push_back("Rare weapon with low damage may be underpowered");
124 result.qualityScore -= 5.0f;
125 }
126
127 // Check weight realism
128 if (item.weight < 100)
129 {
130 result.warnings.push_back("Weapon weight seems too light");
131 result.qualityScore -= 5.0f;
132 }
133 else if (item.weight > 8000)
134 {
135 result.warnings.push_back("Weapon weight seems too heavy");
136 result.qualityScore -= 5.0f;
137 }
138
139 // Clamp quality score
140 result.qualityScore = std::max(0.0f, std::min(100.0f, result.qualityScore));
141
142 return result;
143 }

145 QualityResult CheckWeaponComponentQuality(const ItemWeaponComponentData& item)
146 {
147 QualityResult result;
148 result.isValid = true;
149 result.qualityScore = 100.0f;
150
151 // Check for unrealistic modifiers
152 if (std::abs(item.damageModifier) > 30)
153 {
154 result.warnings.push_back("Very high damage modifier (most components don't affect damage)");
155 result.qualityScore -= 15.0f;
156 }
157
158 // Check component type specific issues
159 std::string typeLower = item.componentType;
160 std::transform(typeLower.begin(), typeLower.end(), typeLower.begin(), ::tolower);
161
162 if (typeLower.find("scope") != std::string::npos || typeLower.find("sight") != std::string::npos)
163 {
164 if (item.damageModifier != 0)
165 {
166 result.errors.push_back("Scopes/Sights cannot modify damage");
167 result.isValid = false;
168 result.qualityScore -= 30.0f;
169 }
170 if (item.recoilModifier != 0)
171 {
172 result.warnings.push_back("Scopes/Sights typically don't modify recoil");
173 result.qualityScore -= 10.0f;
174 }
175 }
176
177 if (typeLower.find("magazine") != std::string::npos)
178 {
179 if (item.damageModifier != 0)
180 {
181 result.warnings.push_back("Magazines should not modify damage");
182 result.qualityScore -= 15.0f;
183 }
184 if (item.penetrationModifier != 0)
185 {
186 result.warnings.push_back("Magazines should not modify penetration");
187 result.qualityScore -= 10.0f;
188 }
189
190 // Check magazine capacity
191 if (item.magazineCapacity < 5)
192 {
193 result.warnings.push_back("Magazine capacity seems very low");
194 result.qualityScore -= 5.0f;
195 }
196 else if (item.magazineCapacity > 150)
197 {
198 result.warnings.push_back("Magazine capacity seems very high");
199 result.qualityScore -= 5.0f;
200 }
201 }
202
203 // Check weight modifier realism
204 if (item.weightModifier > 1000)
205 {
206 result.warnings.push_back("Very high weight modifier");
207 result.qualityScore -= 10.0f;
208 }
209
210 // Check rarity vs modifier correlation
211 if (item.rarity == "Common" && (std::abs(item.accuracyModifier) > 15 || std::abs(item.recoilModifier) > 15))
212 {
213 result.warnings.push_back("Common component with high modifiers may be unbalanced");
214 result.qualityScore -= 5.0f;
215 }
216
217 // Clamp quality score
218 result.qualityScore = std::max(0.0f, std::min(100.0f, result.qualityScore));
219
220 return result;
221 }

388 void PrintQualityResult(const QualityResult& result, const std::string& itemId)
389 {
390 if (result.errors.empty() && result.warnings.empty() && result.qualityScore >= 90.0f)
391 {
392 return; // Skip printing if quality is good
393 }
394
395 std::cout << "[QualityChecker] Item: " << itemId << " (Score: "
396 << std::fixed << std::setprecision(1) << result.qualityScore << "/100)\n";
397
398 for (const auto& error : result.errors)
399 {
400 std::cout << " [ERROR] " << error << "\n";
401 }
402
403 for (const auto& warning : result.warnings)
404 {
405 std::cout << " [WARNING] " << warning << "\n";
406 }
