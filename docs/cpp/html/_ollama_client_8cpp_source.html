26 // Build JSON request body for Ollama API
27 std::string BuildJsonRequest(const std::string& modelName, const std::string& prompt)
28 {
29 std::ostringstream json;
30 json << "{\"model\":\"" << modelName << "\",\"prompt\":\"";
31
32 // Escape JSON special characters in prompt
33 for (char c : prompt)
34 {
35 if (c == '"')
36 json << "\\\"";
37 else if (c == '\\')
38 json << "\\\\";
39 else if (c == '\n')
40 json << "\\n";
41 else if (c == '\r')
42 json << "\\r";
43 else if (c == '\t')
44 json << "\\t";
45 else
46 json << c;
47 }
48
49 json << "\",\"stream\":false}";
50 return json.str();
51 }
52
53 // Extract response text from Ollama JSON response
54 // Ollama API can return multiple JSON objects (one per line) when stream: false
55 // Each line: {"response": "text_chunk", "done": false}
56 // Last line: {"response": "text_chunk", "done": true}
57 std::string ExtractResponseFromJson(const std::string& jsonResponse)
58 {
59 std::string result;
60 std::istringstream stream(jsonResponse);
61 std::string line;
62
63 // Process each line (each line is a JSON object)
64 while (std::getline(stream, line))
65 {
66 // Skip empty lines
67 if (line.empty() || line.find_first_not_of(" \t\r\n") == std::string::npos)
68 continue;
69
70 // Find "response" field in this line
71 size_t responseStart = line.find("\"response\":\"");
72 if (responseStart == std::string::npos)
73 {
74 // Try without quotes (in case of unquoted value, though unlikely)
75 responseStart = line.find("\"response\":");
76 if (responseStart == std::string::npos)
77 {
78 continue; // Skip this line if no response field
79 }
80 responseStart += 11; // Skip "response":
81
82 // Skip whitespace
83 while (responseStart < line.length() &&
84 (line[responseStart] == ' ' || line[responseStart] == '\t'))
85 {
86 responseStart++;
87 }
88
89 // Check if it's a quoted string
90 if (responseStart < line.length() && line[responseStart] == '"')
91 {
92 responseStart++; // Skip opening quote
93 }
94 }
95 else
96 {
97 responseStart += 12; // Skip "response":"
98 }
99
100 // Find the end of the response string value
101 size_t responseEnd = responseStart;
102 bool escaped = false;
103
104 while (responseEnd < line.length())
105 {
106 if (escaped)
107 {
108 escaped = false;
109 responseEnd++;
110 continue;
111 }
112
113 if (line[responseEnd] == '\\')
114 {
115 escaped = true;
116 responseEnd++;
117 continue;
118 }
119
120 if (line[responseEnd] == '"')
121 {
122 break;
123 }
124
125 responseEnd++;
126 }
127
128 if (responseEnd > responseStart)
129 {
130 std::string extracted = line.substr(responseStart, responseEnd - responseStart);
131
132 // Unescape JSON escape sequences and append to result
133 for (size_t i = 0; i < extracted.length(); ++i)
134 {
135 if (extracted[i] == '\\' && i + 1 < extracted.length())
136 {
137 switch (extracted[i + 1])
138 {
139 case 'n':
140 result += '\n';
141 i++;
142 break;
143 case 'r':
144 result += '\r';
145 i++;
146 break;
147 case 't':
148 result += '\t';
149 i++;
150 break;
151 case '\\':
152 result += '\\';
153 i++;
154 break;
155 case '"':
156 result += '"';
157 i++;
158 break;
159 default:
160 result += extracted[i];
161 break;
162 }
163 }
164 else
165 {
166 result += extracted[i];
167 }
168 }
169 }
170 }
171
172 // If we couldn't extract anything, return the original response
173 // (might be that the response is already the final text)
174 if (result.empty())
175 {
176 return jsonResponse;
177 }
178
179 return result;
180 }

183std::string OllamaClient::RunSimple(const std::string& modelName,
184 const std::string& prompt,
185 const OllamaSettings& settings)
186{
187 const std::string host = settings.host.empty() ? "localhost" : settings.host;
188 const INTERNET_PORT port = static_cast<INTERNET_PORT>(settings.port > 0 ? settings.port : 11434);
189 const DWORD requestTimeoutMs = static_cast<DWORD>((settings.requestTimeoutSeconds > 0 ? settings.requestTimeoutSeconds : 120) * 1000);
190 const DWORD connectTimeout = static_cast<DWORD>(settings.connectTimeoutMs > 0 ? settings.connectTimeoutMs : 5000);
191 const DWORD sendTimeout = static_cast<DWORD>(settings.sendTimeoutMs > 0 ? settings.sendTimeoutMs : requestTimeoutMs);
192 const DWORD receiveTimeout = static_cast<DWORD>(settings.receiveTimeoutMs > 0 ? settings.receiveTimeoutMs : requestTimeoutMs);
193
194 std::cout << "[OllamaClient] Calling Ollama HTTP API (model=" << modelName
195 << ", host=" << host << ", port=" << port << ")\n";
196
197 // Build JSON request
198 std::string jsonRequest = BuildJsonRequest(modelName, prompt);
199
200 // Convert host to wide string
201 int wideLen = MultiByteToWideChar(CP_UTF8, 0, host.c_str(), -1, nullptr, 0);
202 std::wstring hostW;
203 hostW.resize(wideLen);
204 MultiByteToWideChar(CP_UTF8, 0, host.c_str(), -1, &hostW[0], wideLen);
205 if (!hostW.empty() && hostW.back() == L'\0')
206 {
207 hostW.pop_back();
208 }
209
210 auto requestStart = std::chrono::steady_clock::now();
211
212 // Initialize WinHTTP
213 HINTERNET hSession = WinHttpOpen(L"RundeeItemFactory/1.0",
214 WINHTTP_ACCESS_TYPE_DEFAULT_PROXY,
215 WINHTTP_NO_PROXY_NAME,
216 WINHTTP_NO_PROXY_BYPASS, 0);
217
218 if (!hSession)
219 {
220 std::cerr << "[OllamaClient] WinHttpOpen failed. Error: " << GetLastError() << "\n";
221 return {};
222 }
223
224 // Connect to Ollama server
225 HINTERNET hConnect = WinHttpConnect(hSession, hostW.c_str(), port, 0);
226
227 if (!hConnect)
228 {
229 std::cerr << "[OllamaClient] WinHttpConnect failed. Error: " << GetLastError() << "\n";
230 WinHttpCloseHandle(hSession);
231 return {};
232 }
233
234 // Open request
235 HINTERNET hRequest = WinHttpOpenRequest(hConnect,
236 L"POST",
237 L"/api/generate",
238 NULL,
239 WINHTTP_NO_REFERER,
240 WINHTTP_DEFAULT_ACCEPT_TYPES,
241 0);
242
243 if (!hRequest)
244 {
245 std::cerr << "[OllamaClient] WinHttpOpenRequest failed. Error: " << GetLastError() << "\n";
246 WinHttpCloseHandle(hConnect);
247 WinHttpCloseHandle(hSession);
248 return {};
249 }
250
251 // Set headers
252 std::wstring headers = L"Content-Type: application/json\r\n";
253 if (!WinHttpAddRequestHeaders(hRequest, headers.c_str(), -1, WINHTTP_ADDREQ_FLAG_ADD))
254 {
255 std::cerr << "[OllamaClient] WinHttpAddRequestHeaders failed. Error: " << GetLastError() << "\n";
256 WinHttpCloseHandle(hRequest);
257 WinHttpCloseHandle(hConnect);
258 WinHttpCloseHandle(hSession);
259 return {};
260 }
261
262 // Set timeout values
263 WinHttpSetTimeouts(hRequest, connectTimeout, connectTimeout, sendTimeout, receiveTimeout);
264
265 // Send request
266 if (!WinHttpSendRequest(hRequest,
267 WINHTTP_NO_ADDITIONAL_HEADERS, 0,
268 (LPVOID)jsonRequest.c_str(),
269 (DWORD)jsonRequest.length(),
270 (DWORD)jsonRequest.length(), 0))
271 {
272 std::cerr << "[OllamaClient] WinHttpSendRequest failed. Error: " << GetLastError() << "\n";
273 WinHttpCloseHandle(hRequest);
274 WinHttpCloseHandle(hConnect);
275 WinHttpCloseHandle(hSession);
276 return {};
277 }
278
279 // Receive response
280 if (!WinHttpReceiveResponse(hRequest, NULL))
281 {
282 std::cerr << "[OllamaClient] WinHttpReceiveResponse failed. Error: " << GetLastError() << "\n";
283 WinHttpCloseHandle(hRequest);
284 WinHttpCloseHandle(hConnect);
285 WinHttpCloseHandle(hSession);
286 return {};
287 }
288
289 // Check status code
290 DWORD statusCode = 0;
291 DWORD statusCodeSize = sizeof(statusCode);
292 if (!WinHttpQueryHeaders(hRequest,
293 WINHTTP_QUERY_STATUS_CODE | WINHTTP_QUERY_FLAG_NUMBER,
294 WINHTTP_HEADER_NAME_BY_INDEX,
295 &statusCode,
296 &statusCodeSize,
297 WINHTTP_NO_HEADER_INDEX))
298 {
299 std::cerr << "[OllamaClient] WinHttpQueryHeaders failed. Error: " << GetLastError() << "\n";
300 WinHttpCloseHandle(hRequest);
301 WinHttpCloseHandle(hConnect);
302 WinHttpCloseHandle(hSession);
303 return {};
304 }
305
306 if (statusCode != 200)
307 {
308 std::cerr << "[OllamaClient] HTTP request failed with status code: " << statusCode << "\n";
309 WinHttpCloseHandle(hRequest);
310 WinHttpCloseHandle(hConnect);
311 WinHttpCloseHandle(hSession);
312 return {};
313 }
314
315 // Read response data
316 std::string response;
317 DWORD bytesAvailable = 0;
318 DWORD bytesRead = 0;
319 char buffer[8192];
320
321 do
322 {
323 if (!WinHttpQueryDataAvailable(hRequest, &bytesAvailable))
324 {
325 std::cerr << "[OllamaClient] WinHttpQueryDataAvailable failed. Error: " << GetLastError() << "\n";
326 break;
327 }
328
329 if (bytesAvailable == 0)
330 break;
331
332 if (!WinHttpReadData(hRequest, buffer, sizeof(buffer) - 1, &bytesRead))
333 {
334 std::cerr << "[OllamaClient] WinHttpReadData failed. Error: " << GetLastError() << "\n";
335 break;
336 }
337
338 if (bytesRead > 0)
339 {
340 buffer[bytesRead] = '\0';
341 response.append(buffer, bytesRead);
342 }
343 } while (bytesRead > 0);
344
345 // Clean up
346 WinHttpCloseHandle(hRequest);
347 WinHttpCloseHandle(hConnect);
348 WinHttpCloseHandle(hSession);
349
350 if (response.empty())
351 {
352 std::cerr << "[OllamaClient] Received empty response.\n";
353 return {};
354 }
355
356 // Extract the actual response text from Ollama JSON
357 std::string extractedResponse = ExtractResponseFromJson(response);
358
359 // Clean up the response (remove any leading/trailing whitespace)
360 std::string trimmed = extractedResponse;
361 size_t start = trimmed.find_first_not_of(" \t\r\n");
362 if (start != std::string::npos)
363 {
364 trimmed = trimmed.substr(start);
365 }
366
367 size_t end = trimmed.find_last_not_of(" \t\r\n");
368 if (end != std::string::npos)
369 {
370 trimmed = trimmed.substr(0, end + 1);
371 }
372
373 // Try to find JSON array/object in the response
374 size_t firstBracket = trimmed.find_first_of("[{");
375 if (firstBracket != std::string::npos)
376 {
377 if (firstBracket > 0)
378 {
379 std::cout << "[OllamaClient] Trimmed " << firstBracket
380 << " leading chars before JSON start.\n";
381 }
382 trimmed = trimmed.substr(firstBracket);
383 }
384
385 size_t lastBracket = trimmed.find_last_of("]}");
386 if (lastBracket != std::string::npos && lastBracket + 1 < trimmed.size())
387 {
388 size_t trimmedCount = trimmed.size() - (lastBracket + 1);
389 if (trimmedCount > 0)
390 {
391 std::cout << "[OllamaClient] Trimmed " << trimmedCount
392 << " trailing chars after JSON end.\n";
393 trimmed = trimmed.substr(0, lastBracket + 1);
394 }
395 }
396
397 auto requestEnd = std::chrono::steady_clock::now();
398 double durationSeconds = std::chrono::duration<double>(requestEnd - requestStart).count();
399 std::cout << "[OllamaClient] HTTP call succeeded in " << std::fixed << std::setprecision(2)
400 << durationSeconds << "s (" << host << ":" << port << ")\n";
401
402 return trimmed;
403}

405std::string OllamaClient::RunWithRetry(const std::string& modelName,
406 const std::string& prompt,
407 int maxRetries,
408 int timeoutSeconds)
409{
410 const OllamaSettings& config = AppConfig::GetOllamaSettings();
411 OllamaSettings effective = config;
412
413 if (timeoutSeconds > 0)
414 {
415 effective.requestTimeoutSeconds = timeoutSeconds;
416 }
417 if (maxRetries > 0)
418 {
419 effective.maxRetries = maxRetries;
420 }
421
422 if (effective.maxRetries <= 0)
423 effective.maxRetries = 3;
424 if (effective.requestTimeoutSeconds <= 0)
425 effective.requestTimeoutSeconds = 120;
426
427 std::string result;
428
429 for (int attempt = 1; attempt <= effective.maxRetries; ++attempt)
430 {
431 std::cout << "[OllamaClient] Attempt " << attempt << " of " << effective.maxRetries << "\n";
432
433 result = RunSimple(modelName, prompt, effective);
434
435 // Check if result is valid (not empty)
436 if (!result.empty())
437 {
438 // Check if result looks like valid JSON (starts with [ or {)
439 std::string trimmed = result;
440 size_t start = trimmed.find_first_not_of(" \t\r\n");
441 if (start != std::string::npos)
442 {
443 trimmed = trimmed.substr(start);
444 if (trimmed[0] == '[' || trimmed[0] == '{')
445 {
446 std::cout << "[OllamaClient] Successfully received response on attempt " << attempt << "\n";
447 return result;
448 }
449 else
450 {
451 std::string preview = trimmed.substr(0, 200);
452 std::cout << "[OllamaClient] Received non-JSON response (preview): " << preview << "\n";
453 }
454 }
455 else
456 {
457 std::cout << "[OllamaClient] Response contained only whitespace.\n";
458 }
459 }
460 else
461 {
462 std::cout << "[OllamaClient] Response was empty.\n";
463 }
464
465 // If this is not the last attempt, wait before retrying
466 if (attempt < effective.maxRetries)
467 {
468 int waitSeconds = 2 * attempt; // Exponential backoff: 2, 4, 6 seconds
469 std::cout << "[OllamaClient] Response invalid or empty. Retrying in " << waitSeconds << " seconds...\n";
470 std::this_thread::sleep_for(std::chrono::seconds(waitSeconds));
471 }
472 else
473 {
474 std::cerr << "[OllamaClient] All " << maxRetries << " attempts failed.\n";
475 }
476 }
477
478 return {}; // Return empty string if all attempts failed
479}
