19 bool LoadPresetFromFile(const std::string& filePath, CustomPreset& preset)
20 {
21 std::ifstream file(filePath);
22 if (!file.is_open())
23 {
24 std::cerr << "[CustomPresetManager] Failed to open preset file: " << filePath << "\n";
25 return false;
26 }
27
28 std::string jsonText((std::istreambuf_iterator<char>(file)),
29 std::istreambuf_iterator<char>());
30 file.close();
31
32 if (jsonText.empty())
33 {
34 std::cerr << "[CustomPresetManager] Preset file is empty: " << filePath << "\n";
35 return false;
36 }
37
38 try
39 {
40 nlohmann::json json = nlohmann::json::parse(jsonText);
41
42 preset.id = JsonUtils::GetStringSafe(json, "id", "");
43 preset.displayName = JsonUtils::GetStringSafe(json, "displayName", "");
44 preset.description = JsonUtils::GetStringSafe(json, "description", "");
45 preset.flavorText = JsonUtils::GetStringSafe(json, "flavorText", "");
46 preset.author = JsonUtils::GetStringSafe(json, "author", "");
47 preset.version = JsonUtils::GetStringSafe(json, "version", "");
48
49 // Load tags array
50 preset.tags.clear();
51 if (json.contains("tags") && json["tags"].is_array())
52 {
53 for (const auto& tag : json["tags"])
54 {
55 if (tag.is_string())
56 {
57 preset.tags.push_back(tag.get<std::string>());
58 }
59 }
60 }
61
62 // Validate
63 std::string errorMsg;
64 if (!ValidatePreset(preset, errorMsg))
65 {
66 std::cerr << "[CustomPresetManager] Invalid preset: " << errorMsg << "\n";
67 return false;
68 }
69
70 return true;
71 }
72 catch (const std::exception& e)
73 {
74 std::cerr << "[CustomPresetManager] Failed to parse preset JSON: " << e.what() << "\n";
75 return false;
76 }
77 }

79 bool SavePresetToFile(const CustomPreset& preset, const std::string& filePath)
80 {
81 std::string errorMsg;
82 if (!ValidatePreset(preset, errorMsg))
83 {
84 std::cerr << "[CustomPresetManager] Cannot save invalid preset: " << errorMsg << "\n";
85 return false;
86 }
87
88 try
89 {
90 nlohmann::json json;
91 json["id"] = preset.id;
92 json["displayName"] = preset.displayName;
93 json["description"] = preset.description;
94 json["flavorText"] = preset.flavorText;
95
96 if (!preset.author.empty())
97 {
98 json["author"] = preset.author;
99 }
100 if (!preset.version.empty())
101 {
102 json["version"] = preset.version;
103 }
104 if (!preset.tags.empty())
105 {
106 json["tags"] = preset.tags;
107 }
108
109 std::ofstream file(filePath);
110 if (!file.is_open())
111 {
112 std::cerr << "[CustomPresetManager] Failed to create preset file: " << filePath << "\n";
113 return false;
114 }
115
116 file << json.dump(2); // Pretty print with 2-space indent
117 file.close();
118
119 return true;
120 }
121 catch (const std::exception& e)
122 {
123 std::cerr << "[CustomPresetManager] Failed to save preset: " << e.what() << "\n";
124 return false;
125 }

149 bool ValidatePreset(const CustomPreset& preset, std::string& errorMessage)
150 {
151 if (preset.id.empty())
152 {
153 errorMessage = "Preset ID is required";
154 return false;
155 }
156
157 // Validate ID format (lowercase, underscore, alphanumeric)
158 for (char c : preset.id)
159 {
160 if (!std::isalnum(c) && c != '_')
161 {
162 errorMessage = "Preset ID must contain only lowercase letters, numbers, and underscores";
163 return false;
164 }
165 if (std::isupper(c))
166 {
167 errorMessage = "Preset ID must be lowercase";
168 return false;
169 }
170 }
171
172 if (preset.displayName.empty())
173 {
174 errorMessage = "Display name is required";
175 return false;
176 }
177
178 if (preset.flavorText.empty())
179 {
180 errorMessage = "Flavor text is required";
181 return false;
182 }
183
184 if (preset.flavorText.length() < 20)
185 {
186 errorMessage = "Flavor text is too short (minimum 20 characters)";
187 return false;
188 }
189
190 return true;
191 }
