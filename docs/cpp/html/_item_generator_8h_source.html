25 /**
26 * @brief Generate items of a single type using LLM
27 *
28 * This function generates items of the specified type using the configured LLM model.
29 * It handles prompt building, LLM communication, JSON parsing, validation, quality checking,
30 * and file writing. Supports retry logic and quality-based regeneration.
31 *
32 * @param args Command line arguments containing generation parameters:
33 * - itemType: Type of items to generate (Food, Drink, Material, Weapon, WeaponComponent, Ammo)
34 * - count: Number of items to generate
35 * - modelName: LLM model name (e.g., "llama3", "mistral")
36 * - preset: Preset type (Default, Forest, Desert, Coast, City) or custom preset
37 * - outputPath: Path to output JSON file
38 * - params: Item-specific generation parameters (maxHunger, maxThirst, etc.)
39 *
40 * @return 0 on success, non-zero on failure
41 *
42 * @note The function automatically merges new items with existing items in the output file,
43 * skipping duplicates based on item ID.
44 *
45 * @see GenerateBatch for generating multiple item types in one operation
46 */
47 int GenerateWithLLM(const CommandLineArgs& args);
48
49 /**
50 * @brief Generate multiple item types in batch mode
51 *
52 * Generates items for multiple types sequentially. Each item type is processed independently
53 * with its own output file (if specified) or default location.
54 *
55 * @param args Command line arguments containing batch configuration:
56 * - batchItems: Vector of BatchItem structures, each specifying:
57 * * itemType: Type of items to generate
58 * * count: Number of items to generate for this type
59 * * outputPath: Optional custom output path (empty uses default)
60 * - modelName: LLM model name (shared across all batch items)
61 * - preset: Preset type (shared across all batch items)
62 *
63 * @return 0 if all batch items succeeded, 1 if any failed
64 *
65 * @note Batch items are processed sequentially, not in parallel.
66 * @note Each batch item can have its own output file or use default location.
67 *
68 * @see GenerateWithLLM for single item type generation
69 */
70 int GenerateBatch(const CommandLineArgs& args);
71}
