25static void EnsureDrinkShape(ItemDrinkData& item)
26{
27 // Default category: Drink
28 if (item.category.empty())
29 item.category = "Drink";
30
31 // Convert lowercase to proper case
32 if (item.category == "drink" || item.category == "DRINK")
33 item.category = "Drink";
34
35 // Drink items prioritize thirstRestore
36 if (item.thirstRestore < 5)
37 item.thirstRestore = 5;
38
39 // Drink items should have thirstRestore greater than hungerRestore
40 if (item.thirstRestore < item.hungerRestore)
41 {
42 item.thirstRestore = item.hungerRestore + 5;
43 }
44}

46static void EnsureSpoilage(ItemDrinkData& item)
47{
48 // If spoils == false, spoilTimeMinutes should be 0
49 if (!item.spoils)
50 {
51 item.spoilTimeMinutes = 0;
52 return;
53 }
54
55 // If spoils == true, set min/max range (5 minutes ~ 7 days)
56 int minMinutes = 5;
57 int maxMinutes = 7 * 24 * 60; // 7 days
58 item.spoilTimeMinutes = JsonUtils::ClampInt(item.spoilTimeMinutes, minMinutes, maxMinutes);
59}

61static void EnsureRarity(ItemDrinkData& item)
62{
63 float power = ComputePower(item);
64
65 // If rarity is empty or invalid, auto-classify based on power
66 if (item.rarity.empty() ||
67 (item.rarity != "Common" && item.rarity != "Uncommon" && item.rarity != "Rare"))
68 {
69 if (power <= 25.0f) item.rarity = "Common";
70 else if (power <= 55.0f) item.rarity = "Uncommon";
71 else item.rarity = "Rare";
72 return;
73 }
74
75 // Adjust to match rarity range
76 float maxPower = 999.0f;
77 if (item.rarity == "Common")
78 {
79 maxPower = 40.0f;
80 }
81 else if (item.rarity == "Uncommon")
82 {
83 maxPower = 65.0f;
84 }
85 else if (item.rarity == "Rare")
86 {
87 maxPower = 100.0f;
88 }
89
90 if (power <= maxPower)
91 return; // Already within range
92
93 // If power is too high, scale down while maintaining ratio
94 float scale = maxPower / power;
95 if (scale <= 0.0f)
96 return;
97
98 auto scaleAndClamp = [&](int& v)
99 {
100 int scaled = static_cast<int>(v * scale + 0.5f);
101 v = JsonUtils::ClampInt(scaled, 0, 100);
102 };
103
104 scaleAndClamp(item.hungerRestore);
105 scaleAndClamp(item.thirstRestore);
106 scaleAndClamp(item.healthRestore);
107}

109void DrinkItemValidator::Validate(ItemDrinkData& item)
110{
111 std::string originalId = item.id;
112
113 // 0) Normalize prefix: strip any leading drink_ (any case, repeated), then add canonical "Drink_"
114 if (!item.id.empty())
115 {
116 auto stripPrefix = [](std::string& value, const std::string& prefixLower)
117 {
118 std::string lower = value;
119 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
120 while (lower.rfind(prefixLower, 0) == 0)
121 {
122 value = value.substr(prefixLower.size());
123 lower = value;
124 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
125 }
126 };
127
128 stripPrefix(item.id, "drink_");
129 item.id = "Drink_" + item.id;
130 }
131
132 // Debug logging disabled for release
133
134 // 1) Clamp basic values
135 item.hungerRestore = JsonUtils::ClampInt(item.hungerRestore, 0, 100);
136 item.thirstRestore = JsonUtils::ClampInt(item.thirstRestore, 0, 100);
137 item.healthRestore = JsonUtils::ClampInt(item.healthRestore, 0, 100);
138
139 item.maxStack = JsonUtils::ClampInt(item.maxStack, 1, 999);
140
141 // 2) Ensure Drink characteristics (thirstRestore priority)
142 EnsureDrinkShape(item);
143
144 // 3) Handle spoilage
145 EnsureSpoilage(item);
146
147 // 4) Balance rarity
148 EnsureRarity(item);
149
150 // 5) Ensure description is not empty
151 if (item.description.empty())
152 {
153 item.description = "A " + item.displayName + " that restores thirst.";
154 std::cout << "[DrinkItemValidator] Warning: Item " << item.id << " has empty description, using default.\n";
155 }
156}
