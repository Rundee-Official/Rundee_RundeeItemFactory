18/**
19 * @struct CustomPreset
20 * @brief Data structure for custom world presets
21 *
22 * Represents a user-defined preset with custom world context and flavor text.
23 * Used instead of built-in presets to provide unique world settings for item generation.
24 */
25struct CustomPreset
26{
27 /**
28 * @brief Unique identifier (lowercase, underscore-separated)
29 *
30 * Examples: "post_apocalyptic_city", "fantasy_forest"
31 */
32 std::string id;
33
34 /**
35 * @brief Human-readable display name
36 *
37 * Examples: "Post-Apocalyptic City", "Fantasy Forest"
38 */
39 std::string displayName;
40
41 /**
42 * @brief Description of the preset
43 *
44 * Explains what this preset represents and when to use it.
45 */
46 std::string description;
47
48 /**
49 * @brief World context text for LLM prompts
50 *
51 * This text is included in prompts to guide LLM generation with the
52 * appropriate world context and flavor.
53 */
54 std::string flavorText;
55
56 /**
57 * @brief Optional author name
58 *
59 * Creator of the preset (optional metadata).
60 */
61 std::string author;
62
63 /**
64 * @brief Optional version string
65 *
66 * Version of the preset (optional metadata).
67 */
68 std::string version;
69
70 /**
71 * @brief Optional tags for categorization
72 *
73 * Examples: "survival", "post-apocalyptic", "fantasy", "sci-fi"
74 */
75 std::vector<std::string> tags;
76
77 /**
78 * @brief Check if preset is valid
79 *
80 * A preset is valid if it has a non-empty id, displayName, and flavorText.
81 *
82 * @return true if preset is valid, false otherwise
83 */
84 bool isValid() const
85 {
86 return !id.empty() && !displayName.empty() && !flavorText.empty();
87 }
88};
