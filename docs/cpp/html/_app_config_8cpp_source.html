60void AppConfig::LoadFromDefaultLocation()
61{
62 if (g_loaded)
63 return;
64
65 try
66 {
67 std::filesystem::path base = std::filesystem::current_path();
68 std::filesystem::path defaultPath = base / "config" / "rundee_config.json";
69
70 if (!std::filesystem::exists(defaultPath))
71 {
72 // No config file is fine; keep defaults.
73 g_loaded = true;
74 return;
75 }
76
77 std::string jsonText = ReadAllText(defaultPath);
78 if (jsonText.empty())
79 {
80 std::cerr << "[AppConfig] Config file is empty: " << defaultPath.string() << "\n";
81 g_loaded = true;
82 return;
83 }
84
85 ApplyJson(jsonText, defaultPath.string());
86 g_loaded = true;
87 g_loadedPath = defaultPath.string();
88 }
89 catch (const std::exception& ex)
90 {
91 std::cerr << "[AppConfig] Failed to load default config: " << ex.what() << "\n";
92 g_loaded = true;
93 }
94}

125void AppConfig::ApplyJson(const std::string& jsonText, const std::string& sourceLabel)
126{
127 try
128 {
129 json root = json::parse(jsonText);
130
131 if (root.contains("ollama") && root["ollama"].is_object())
132 {
133 const auto& o = root["ollama"];
134 SetIfPresent(o, "host", g_settings.host);
135 SetIfPresent(o, "port", g_settings.port);
136 SetIfPresent(o, "maxRetries", g_settings.maxRetries);
137 SetIfPresent(o, "requestTimeoutSeconds", g_settings.requestTimeoutSeconds);
138 SetIfPresent(o, "connectTimeoutMs", g_settings.connectTimeoutMs);
139 SetIfPresent(o, "sendTimeoutMs", g_settings.sendTimeoutMs);
140 SetIfPresent(o, "receiveTimeoutMs", g_settings.receiveTimeoutMs);
141 }
142
143 std::cout << "[AppConfig] Loaded config from " << sourceLabel << "\n";
144 }
145 catch (const std::exception& ex)
146 {
147 std::cerr << "[AppConfig] Failed to parse config (" << sourceLabel << "): " << ex.what() << "\n";
148 }
149}
