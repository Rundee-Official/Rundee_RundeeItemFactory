17std::string PromptTemplateLoader::GetTemplateDirectory()
18{
19 // Get executable directory
20 char exePath[MAX_PATH];
21 DWORD pathLen = GetModuleFileNameA(NULL, exePath, MAX_PATH);
22 if (pathLen == 0 || pathLen >= MAX_PATH)
23 {
24 return "prompts/"; // Fallback to relative path
25 }
26
27 // Extract directory
28 std::string exeDir = exePath;
29 size_t lastSlash = exeDir.find_last_of("\\/");
30 if (lastSlash != std::string::npos)
31 {
32 exeDir = exeDir.substr(0, lastSlash + 1);
33 }
34
35 return exeDir + "prompts/";
36}

45std::string PromptTemplateLoader::LoadTemplate(const std::string& templateName,
46 const std::string& presetContext,
47 int maxHunger,
48 int maxThirst,
49 int count,
50 const std::set<std::string>& excludeIds,
51 const std::string& presetName,
52 const std::string& itemTypeName,
53 const std::string& modelName,
54 const std::string& generationTimestamp,
55 int existingCount)
56{
57 std::string templatePath = GetTemplateDirectory() + templateName + ".txt";
58
59 // Try to read template file
60 std::ifstream file(templatePath);
61 if (!file.is_open())
62 {
63 // Template file not found - return empty string (will fall back to hardcoded prompts)
64 return {};
65 }
66
67 // Read entire file
68 std::stringstream buffer;
69 buffer << file.rdbuf();
70 std::string templateContent = buffer.str();
71 file.close();
72
73 // Replace variables
74 // {PRESET_CONTEXT}
75 size_t pos = templateContent.find("{PRESET_CONTEXT}");
76 while (pos != std::string::npos)
77 {
78 templateContent.replace(pos, 16, presetContext);
79 pos = templateContent.find("{PRESET_CONTEXT}", pos + presetContext.length());
80 }
81
82 // {PRESET_NAME}
83 if (!presetName.empty())
84 {
85 pos = templateContent.find("{PRESET_NAME}");
86 while (pos != std::string::npos)
87 {
88 templateContent.replace(pos, 13, presetName);
89 pos = templateContent.find("{PRESET_NAME}", pos + presetName.length());
90 }
91 }
92
93 // {ITEM_TYPE}
94 if (!itemTypeName.empty())
95 {
96 pos = templateContent.find("{ITEM_TYPE}");
97 while (pos != std::string::npos)
98 {
99 templateContent.replace(pos, 10, itemTypeName);
100 pos = templateContent.find("{ITEM_TYPE}", pos + itemTypeName.length());
101 }
102 }
103
104 // {MODEL_NAME}
105 if (!modelName.empty())
106 {
107 pos = templateContent.find("{MODEL_NAME}");
108 while (pos != std::string::npos)
109 {
110 templateContent.replace(pos, 12, modelName);
111 pos = templateContent.find("{MODEL_NAME}", pos + modelName.length());
112 }
113 }
114
115 // {TIMESTAMP}
116 if (!generationTimestamp.empty())
117 {
118 pos = templateContent.find("{TIMESTAMP}");
119 while (pos != std::string::npos)
120 {
121 templateContent.replace(pos, 11, generationTimestamp);
122 pos = templateContent.find("{TIMESTAMP}", pos + generationTimestamp.length());
123 }
124 }
125
126 // {MAX_HUNGER}
127 pos = templateContent.find("{MAX_HUNGER}");
128 while (pos != std::string::npos)
129 {
130 templateContent.replace(pos, 12, std::to_string(maxHunger));
131 pos = templateContent.find("{MAX_HUNGER}", pos + std::to_string(maxHunger).length());
132 }
133
134 // {MAX_THIRST}
135 pos = templateContent.find("{MAX_THIRST}");
136 while (pos != std::string::npos)
137 {
138 templateContent.replace(pos, 12, std::to_string(maxThirst));
139 pos = templateContent.find("{MAX_THIRST}", pos + std::to_string(maxThirst).length());
140 }
141
142 // {COUNT}
143 pos = templateContent.find("{COUNT}");
144 while (pos != std::string::npos)
145 {
146 templateContent.replace(pos, 7, std::to_string(count));
147 pos = templateContent.find("{COUNT}", pos + std::to_string(count).length());
148 }
149
150 // {EXCLUDE_IDS}
151 pos = templateContent.find("{EXCLUDE_IDS}");
152 if (pos != std::string::npos)
153 {
154 std::string excludeIdsText;
155 if (!excludeIds.empty())
156 {
157 excludeIdsText = "\nIMPORTANT - Avoid these existing item IDs (do NOT use these):\n";
158 int idCount = 0;
159 for (const auto& id : excludeIds)
160 {
161 if (idCount > 0) excludeIdsText += ", ";
162 excludeIdsText += id;
163 idCount++;
164 if (idCount >= 20) // Limit to first 20 to avoid prompt bloat
165 {
166 excludeIdsText += " ... (and " + std::to_string(excludeIds.size() - 20) + " more, list truncated)";
167 break;
168 }
169 }
170 excludeIdsText += "\nGenerate NEW unique IDs that are different from all existing IDs (assume many more exist). Avoid reusing stems; use fresh, novel names, not simple number suffixes.\n";
171 }
172 templateContent.replace(pos, 13, excludeIdsText);
173 }
174
175 // {EXISTING_COUNT}
176 if (existingCount >= 0)
177 {
178 pos = templateContent.find("{EXISTING_COUNT}");
179 while (pos != std::string::npos)
180 {
181 templateContent.replace(pos, 16, std::to_string(existingCount));
182 pos = templateContent.find("{EXISTING_COUNT}", pos + std::to_string(existingCount).length());
183 }
184 }
185
186 return templateContent;
187}
