81 /**
82 * @brief Parse command line arguments
83 *
84 * Parses argc/argv into a CommandLineArgs structure. Supports:
85 * - Model selection: --model <name>
86 * - Item type: --type <food|drink|material|weapon|weaponcomponent|ammo>
87 * - Count: --count <number>
88 * - Preset: --preset <default|forest|desert|coast|city>
89 * - Output path: --output <path>
90 * - Batch mode: --batch <type1:count1,type2:count2,...>
91 * - Custom preset: --custom-preset <path>
92 * - Additional prompt: --additional-prompt <text>
93 *
94 * @param argc Number of command line arguments
95 * @param argv Array of command line argument strings
96 * @return Parsed CommandLineArgs structure
97 */
98 CommandLineArgs ParseArguments(int argc, char** argv);
99
100 /**

102 *
103 * Parses a comma-separated batch string like "food:10,weapon:5,ammo:20"

105 *
106 * @param batchStr Batch string in format "type1:count1,type2:count2,..."

109 * @note Item type names are case-insensitive
