21 void Validate(ItemWeaponData& item)
22 {
23 std::string originalId = item.id;
24
25 // Normalize prefix: strip any leading weapon_ (any case, repeated), then add canonical "Weapon_"
26 if (!item.id.empty())
27 {
28 auto stripPrefix = [](std::string& value, const std::string& prefixLower)
29 {
30 std::string lower = value;
31 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
32 while (lower.rfind(prefixLower, 0) == 0)
33 {
34 value = value.substr(prefixLower.size());
35 lower = value;
36 std::transform(lower.begin(), lower.end(), lower.begin(), ::tolower);
37 }
38 };
39
40 stripPrefix(item.id, "weapon_");
41 item.id = "Weapon_" + item.id;
42 }
43
44 // Debug logging disabled for release
45
46 // Default category
47 if (item.category.empty())
48 item.category = "Weapon";
49
50 // Default weaponCategory
51 if (item.weaponCategory.empty())
52 {
53 // Try to infer from weaponType
54 std::string lowerType = item.weaponType;
55 std::transform(lowerType.begin(), lowerType.end(), lowerType.begin(), ::tolower);
56 if (lowerType.find("sword") != std::string::npos ||
57 lowerType.find("axe") != std::string::npos ||
58 lowerType.find("knife") != std::string::npos ||
59 lowerType.find("mace") != std::string::npos ||
60 lowerType.find("spear") != std::string::npos ||
61 lowerType.find("club") != std::string::npos ||
62 lowerType.find("hammer") != std::string::npos ||
63 lowerType.find("blade") != std::string::npos ||
64 lowerType.find("melee") != std::string::npos)
65 {
66 item.weaponCategory = "Melee";
67 }
68 else
69 {
70 item.weaponCategory = "Ranged";
71 }
72 }
73
74 // Clamp stat values
75 item.minDamage = JsonUtils::ClampInt(item.minDamage, 0, 100);
76 item.maxDamage = JsonUtils::ClampInt(item.maxDamage, 0, 100);
77 item.fireRate = JsonUtils::ClampInt(item.fireRate, 0, 1200);
78 item.accuracy = JsonUtils::ClampInt(item.accuracy, 0, 100);
79 item.recoil = JsonUtils::ClampInt(item.recoil, 0, 100);
80 item.ergonomics = JsonUtils::ClampInt(item.ergonomics, 0, 100);
81 item.durability = JsonUtils::ClampInt(item.durability, 0, 100);
82 item.muzzleVelocity = JsonUtils::ClampInt(item.muzzleVelocity, 0, 1500);
83 item.effectiveRange = JsonUtils::ClampInt(item.effectiveRange, 0, 1000);
84 item.penetrationPower = JsonUtils::ClampInt(item.penetrationPower, 0, 100);
85 item.moddingSlots = JsonUtils::ClampInt(item.moddingSlots, 0, 10);
86 item.attackSpeed = JsonUtils::ClampInt(item.attackSpeed, 0, 10);
87 item.reach = JsonUtils::ClampInt(item.reach, 0, 500); // 0-500cm (0-5 meters)
88 item.staminaCost = JsonUtils::ClampInt(item.staminaCost, 0, 100);
89 item.maxStack = JsonUtils::ClampInt(item.maxStack, 1, 999);
90
91 // Validate based on weaponCategory
92 if (item.weaponCategory == "Melee")
93 {
94 // Melee weapons: no caliber, no ranged stats
95 if (!item.caliber.empty())
96 {
97 item.caliber = ""; // Clear caliber for melee
98 }
99 item.muzzleVelocity = 0;
100 item.effectiveRange = 0;
101 item.penetrationPower = 0;
102 item.recoil = 0;
103
104 // Ensure melee stats are set
105 if (item.attackSpeed == 0)
106 item.attackSpeed = 2; // Default 2 attacks per second
107 if (item.reach == 0)
108 item.reach = 100; // Default 1 meter reach
109 }
110 else if (item.weaponCategory == "Ranged")
111 {
112 // Ranged weapons: must have caliber
113 if (item.caliber.empty())
114 {
115 item.caliber = "9mm"; // Default caliber
116 std::cout << "[WeaponItemValidator] Warning: Ranged weapon " << item.id << " has empty caliber, using default.\n";
117 }
118
119 // Melee-only stats should be 0
120 item.attackSpeed = 0;
121 item.reach = 0;
122 item.staminaCost = 0;
123 }
124
125 // Normalize weight to avoid unrealistically light weapons
126 if (item.weight < 1500)
127 {
128 item.weight = 1500;
129 }
130
131 // For ranged, give default muzzle velocity if missing
132 if (item.weaponCategory == "Ranged" && item.muzzleVelocity == 0)
133 {
134 // Simple default based on type
135 std::string lowType = item.weaponType;
136 std::transform(lowType.begin(), lowType.end(), lowType.begin(), ::tolower);
137 if (lowType.find("bow") != std::string::npos || lowType.find("crossbow") != std::string::npos)
138 item.muzzleVelocity = 300;
139 else
140 item.muzzleVelocity = 400;
141 }
142
143 // Ensure maxDamage >= minDamage
144 if (item.maxDamage < item.minDamage)
145 {
146 item.maxDamage = item.minDamage;
147 }
148
149 // Ensure description is not empty
150 if (item.description.empty())
151 {
152 item.description = "A " + item.displayName + " weapon.";
153 std::cout << "[WeaponItemValidator] Warning: Item " << item.id << " has empty description, using default.\n";
154 }
155 }
156}
